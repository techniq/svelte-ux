<Preview showCode>

  <div>{format(1234.56, 'currency')}</div>
  <div>{format(.5678, 'percent')}</div>
  <div>{format(.5678, 'percentRound')}</div>
  <div>{format(1_234_567, 'metric')}</div>
  <div>{format(1_200_000, 'metric')}</div>
  <div>{format(.5678, 'percent', 1)}</div>
</Preview>
